void sort(char a[][3],int n)
{
	int i;
	char ch;
	for(i=0;i<n;i++)
	{
		if(a[i][0]>a[i][1])
		{
			ch=a[i][0];
			a[i][0]=a[i][1];
			a[i][1]=ch;
		}

		{
			ch=a[i][0];
			a[i][0]=a[i][2];

		}

		{
			ch=a[i][1];
			a[i][1]=a[i][2];
			a[i][2]=ch;
		}
	
	}
	
}
